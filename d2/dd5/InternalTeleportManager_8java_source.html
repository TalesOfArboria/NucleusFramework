106  @Override
107  public IScheduledTeleport teleport(Plugin plugin, Player player,
108  Location location, int tickDelay, TeleportMode mode) {
109  PreCon.notNull(plugin);
110  PreCon.notNull(player);
111  PreCon.notNull(location);
112  PreCon.positiveNumber(tickDelay);
113  PreCon.notNull(mode);
114 
115  ScheduledTeleport scheduled = _scheduled.get(player.getUniqueId());
116  if (scheduled != null)
117  scheduled.cancel();
118 
119  location = LocationUtils.copy(location);
120 
121  TeleportScheduledEvent event = new TeleportScheduledEvent(player, location, tickDelay);
122  if (META.has(player, TELEPORT_DENY_META_NAME)) {
123  event.setCancelled(true);
124  }
125 
126  Nucleus.getEventManager().callBukkit(this, event);
127 
128  scheduled = new ScheduledTeleport(this, player, event.getDelayTicks(), location,
129  TeleportCause.PLUGIN, mode);
130 
131  if (event.isCancelled()) {
132  scheduled.cancel();
133  return scheduled;
134  }
135 
136  _scheduled.put(player.getUniqueId(), scheduled);
137  Scheduler.runTaskLater(plugin, scheduled.getDelayTicks(), scheduled);
138 
139  return scheduled;
140  }
