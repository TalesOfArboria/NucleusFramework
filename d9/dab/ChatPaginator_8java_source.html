342  public void show(CommandSender sender, int page, Object format) {
343  PreCon.notNull(sender);
344  PreCon.notNull(format);
345 
346  page = page > 0 ? page : 1;
347 
348  int totalPages = getTotalPages(format);
349 
350  loadCommands(page, totalPages);
351 
352  IChatMessage header = getFormattedHeader(page, totalPages);
353  if (header.length() != 0)
354  _msg.tell(sender, header);
355 
356  if (page < 1 || page > totalPages) {
357  if (page == 1) {
358  _msg.tell(sender, NucLang.get(getPlugin(), _NO_ITEMS));
359  }
360  else {
361  _msg.tell(sender, NucLang.get(getPlugin(), _PAGE_NOT_FOUND, page));
362  }
363  }
364  else if (_searchTerm == null) {
365  showAll(sender, page, format);
366  }
367  else {
368  showSearch(sender, page, format);
369  }
370 
371  IChatMessage footer = getFormattedFooter(page, totalPages);
372  if (footer.length() != 0)
373  _msg.tell(sender, footer);
374  }

464  protected void loadCommands(int page, int totalPages) {
465  if (_commands == null) {
466  _prev = new TextArg("");
467  _next = _prev;
468  return;
469  }
470 
471  String prevCommand = _commands.getPrevCommand(this, page, totalPages);
472  String nextCommand = _commands.getNextCommand(this, page, totalPages);
473 
474  _prev = prevCommand == null
475  ? new TextArg(NucLang.get(_plugin, _PREV_DISABLED))
476  : new TextArg(NucLang.get(_plugin, _PREV),
477  new ClickableArgModifier(ClickAction.RUN_COMMAND, prevCommand),
478  new HoverableArgModifier(HoverAction.SHOW_TEXT, NucLang.get(_plugin, _PREV_HOVER)));
479 
480  _next = nextCommand == null
481  ? new TextArg(NucLang.get(_plugin, _NEXT_DISABLED))
482  : new TextArg(NucLang.get(_plugin, _NEXT),
483  new ClickableArgModifier(ClickAction.RUN_COMMAND, nextCommand),
484  new HoverableArgModifier(HoverAction.SHOW_TEXT, NucLang.get(_plugin, _NEXT_HOVER)));
